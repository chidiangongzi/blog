示例脚本

脚本各行参数 说明
#PBS -N mkr_ys 用于提交作业名称
#PBS -S /bin/bash 提交脚本的语法
#PBS -P P001801003 老师的项目号
#PBS -q Regular 目标队列
#PBS -l nodes=3:ppn=28 申请资源列表3个节点，28个MPI任务/节点
#PBS -l walltime=12:00:00 估计最大运算时间，超时会自动中断作业
#PBS -e jobname.err 报错文件
cd $PBS_O_WORKDIR 进入脚本所在目录
nprocs=`cat $PBS_NODEFILE wc -l`
mpirun -np $nprocs ./wrf.exe
